const express = require('express');
const multer = require('multer');
const AWS = require('aws-sdk');
const KnowledgeBase = require('../models/knowledgeBase');
const router = express.Router();


const s3 = new AWS.S3();
const upload = multer({ dest: 'uploads/' });


router.post('/upload', upload.array('files'), async (req, res) => {
  const files = req.files.map(file => ({
    url: `https://${process.env.AWS_S3_BUCKET}.s3.amazonaws.com/${file.filename}`,
    type: file.mimetype,
    name: file.originalname,
  }));
  for (const file of req.files) {
    await s3.upload({
      Bucket: process.env.AWS_S3_BUCKET,
      Key: file.filename,
      Body: require('fs').createReadStream(file.path),
    }).promise();
  }
  const base = new KnowledgeBase({ ...req.body, files });
  await base.save();
  res.json(base);
});


router.get('/download/:baseId', async (req, res) => {
  const base = await KnowledgeBase.findById(req.params.baseId);
  const zip = require('archiver')('zip');
  res.attachment(`${base.title}.zip`);
  zip.pipe(res);
  for (const file of base.files) {
    const { Body } = await s3.getObject({ Bucket: process.env.AWS_S3_BUCKET, Key: file.name }).promise();
    zip.append(Body, { name: file.name });
  }
  zip.finalize();
});

import React, { useCallback } from 'react';
import { useDropzone } from 'react-dropzone';
import axios from 'axios';
import Card from './Card';
import Button from './Button';


const Uploader = ({ onUpload }) => {
  const onDrop = useCallback(async (acceptedFiles) => {
    const formData = new FormData();
    acceptedFiles.forEach(file => formData.append('files', file));
    formData.append('title', 'New Base');
    formData.append('creator', 'user_id');
    formData.append('subject', 'Math');
    formData.append('price', 10);
    const res = await axios.post('http://localhost:3000/api/upload', formData);
    onUpload(res.data);
  }, []);


  const { getRootProps, getInputProps } = useDropzone({ onDrop });


  return (
    <Card className="p-4">
      <div {...getRootProps()} className="border-dashed border-2 p-4 text-center">
        <input {...getInputProps()} />
        <p>Drag & drop files here, or click to select</p>
      </div>
    </Card>
  );
};

import React, { useState } from 'react';
import { useQuery, gql } from '@apollo/client';
import Card from '../components/Card';
import Button from '../components/Button';
import Uploader from '../components/Uploader';
import axios from 'axios';


const GET_BASES = gql`query { knowledgeBases { id title creator subject price files { url name } } }`;


const Marketplace = () => {
  const { loading, data } = useQuery(GET_BASES);
  const [uploadedBase, setUploadedBase] = useState(null);


  const handleDownload = async (baseId) => {
    const res = await axios.get(`http://localhost:3000/api/upload/download/${baseId}`, { responseType: 'blob' });
    const url = window.URL.createObjectURL(new Blob([res.data]));
    const link = document.createElement('a');
    link.href = url;
    link.setAttribute('download', 'knowledge_base.zip');
    document.body.appendChild(link);
    link.click();
  };


  return (
    <div className="min-h-screen bg-secondary p-8">
      <h1 className="text-3xl font-bold text-primary mb-4">Knowledge Base Marketplace</h1>
      <Uploader onUpload={setUploadedBase} />
      {loading ? <p>Loading...</p> : data.knowledgeBases.map(base => (
        <Card key={base.id} className="mb-4">
          <h2>{base.title}</h2>
          <p>By {base.creator} | ${base.price}</p>
          <Button onClick={() => handleDownload(base.id)}>Download</Button>
        </Card>
      ))}
    </div>
  );
};
