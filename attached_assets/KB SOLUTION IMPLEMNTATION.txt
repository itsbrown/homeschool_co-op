import sys
import json
from elasticsearch import Elasticsearch
import spacy
import requests


es = Elasticsearch(['http://localhost:9200'])
nlp = spacy.load('en_core_web_sm')


def extract_entities(text):
    doc = nlp(text)
    entities = [(ent.text, ent.label_) for ent in doc.ents]
    return entities


def summarize_text(text):
    # Simulated Claude API call for summarization
    response = requests.post('https://your-claude-endpoint', 
                            json={'text': text, 'prompt': 'Summarize the following text for a 6-7 year old audience', 'max_tokens': 100})
    return response.json().get('summary', text[:200])


def generate_questions(summary, entities, subject):
    # Simulated Claude API call for question generation
    entity_names = [e[0] for e in entities if e[1] in ['PERSON', 'ORG', 'GPE', 'EVENT']]
    prompt = f"Generate 3 simple comprehension questions for 6-7 year olds based on this summary: {summary}. Focus on {subject} and mention {', '.join(entity_names)} if relevant."
    response = requests.post('https://your-claude-endpoint', 
                            json={'text': prompt, 'max_tokens': 150})
    questions = response.json().get('questions', [
        {'question': f'What does this tell us about {subject}?', 'answer': 'To be filled by teacher'},
        {'question': f'Who is mentioned in the text?', 'answer': 'To be filled by teacher'},
        {'question': f'What happened in the story?', 'answer': 'To be filled by teacher'}
    ])
    return questions


def generate_activity(age, subject, activity_type, ocr_text=None):
    # Fetch relevant content from the knowledge base
    content = ocr_text if ocr_text else ''
    if not content:
        search_result = es.search(index='knowledge_base', body={
            'query': {
                'bool': {
                    'must': [
                        {'match': {'subject': subject}},
                        {'range': {'difficulty': {'gte': 'easy', 'lte': 'medium'}}}
                    ]
                }
            }
        })
        content = search_result['hits']['hits'][0]['_source']['text'] if search_result['hits']['hits'] else 'Default content'


    # Extract entities and summarize content
    entities = extract_entities(content)
    summary = summarize_text(content)


    # Generate coloring guide for history theme
    coloring_guide = []
    if subject.lower() == 'history' and activity_type == 'coloring':
        coloring_guide = [
            {'element': 'George Washington', 'instruction': 'America’s first president in his general’s uniform.'},
            {'element': 'Liberty Bell', 'instruction': 'Famous bell with a crack, rung for independence.'},
            {'element': 'Original American Flag', 'instruction': 'Flag with 13 stars arranged in a circle.'},
            {'element': 'Independence Hall', 'instruction': 'Building where the Declaration was signed.'},
        ]


    # Generate facts
    facts = [
        'The United States became a country in 1776.',
        'George Washington led American soldiers.',
        'The Liberty Bell is a symbol of freedom.',
        'The first American flag had 13 stars.'
    ]


    # Generate worksheet questions based on knowledge base content
    worksheet = {
        'questions': generate_questions(summary, entities, subject)
    }


    return {
        'coloring_guide': coloring_guide,
        'facts': facts,
        'worksheet': worksheet
    }


if __name__ == '__main__':
    data = json.loads(sys.argv[1])
    result = generate_activity(data['age'], data['subject'], data['activityType'])
