export type PaymentFrequency = 'weekly' | 'biweekly' | 'monthly' | 'one_time';

export interface PaymentSchedule {
  totalAmount: number;
  numberOfPayments: number;
  paymentAmount: number;
  finalPaymentAmount: number;
  paymentDates: Date[];
  frequency: PaymentFrequency;
  startDate: Date;
  endDate: Date;
}

// Helper: distribute remainder evenly across ALL payments (no giant final payment)
function distributeRemainder(base: number, remainder: number, count: number): number[] {
  const amounts = new Array(count).fill(base);
  for (let i = 0; i < remainder; i++) {
    amounts[i % count] += 1; // round-robin cents = visually fair
  }
  return amounts;
}

function daysBetween(start: Date, end: Date): number {
  const msPerDay = 1000 * 60 * 60 * 24;
  const utcStart = Date.UTC(start.getFullYear(), start.getMonth(), start.getDate());
  const utcEnd = Date.UTC(end.getFullYear(), end.getMonth(), end.getDate());
  return Math.floor((utcEnd - utcStart) / msPerDay);
}

function safeParseDate(date: any, fieldName: string): Date | null {
  // (keep your existing safeParseDate function unchanged)
  if (!date) return null;
  let parsed = date instanceof Date ? date : new Date(date);
  return isNaN(parsed.getTime()) ? null : parsed;
}

export function calculatePaymentSchedule(
  totalAmountCents: number,
  startDate: Date,
  endDate: Date,
  frequency: PaymentFrequency
): PaymentSchedule {
  if (totalAmountCents < 50) throw new Error("Amount below Stripe minimum (50 cents)");

  const validStart = safeParseDate(startDate, 'startDate');
  const validEnd = safeParseDate(endDate, 'endDate');
  if (!validStart || !validEnd) {
    const today = new Date();
    return { totalAmount: totalAmountCents, numberOfPayments: 1, paymentAmount: totalAmountCents, finalPaymentAmount: totalAmountCents, paymentDates: [today], frequency: 'one_time', startDate: today, endDate: today };
  }

  if (frequency === 'one_time') {
    return { totalAmount: totalAmountCents, numberOfPayments: 1, paymentAmount: totalAmountCents, finalPaymentAmount: totalAmountCents, paymentDates: [new Date(validStart)], frequency, startDate: new Date(validStart), endDate: new Date(validEnd) };
  }

  const totalDays = daysBetween(validStart, validEnd);
  const intervalDays = frequency === 'weekly' ? 7 : frequency === 'biweekly' ? 14 : 30;

  if (totalDays < intervalDays) {
    const today = new Date(validStart);
    return { totalAmount: totalAmountCents, numberOfPayments: 1, paymentAmount: totalAmountCents, finalPaymentAmount: totalAmountCents, paymentDates: [today], frequency: 'one_time', startDate: today, endDate: new Date(validEnd) };
  }

  const paymentDates: Date[] = [];
  let current = new Date(validStart);
  while (current <= validEnd) {
    paymentDates.push(new Date(current));
    current.setDate(current.getDate() + intervalDays);
  }

  if (paymentDates.length < 2) {
    return { totalAmount: totalAmountCents, numberOfPayments: 1, paymentAmount: totalAmountCents, finalPaymentAmount: totalAmountCents, paymentDates: [new Date(validStart)], frequency: 'one_time', startDate: new Date(validStart), endDate: new Date(validEnd) };
  }

  const numberOfPayments = paymentDates.length;
  const base = Math.floor(totalAmountCents / numberOfPayments);
  const remainder = totalAmountCents - (base * numberOfPayments);
  const paymentAmounts = distributeRemainder(base, remainder, numberOfPayments);

  return {
    totalAmount: totalAmountCents,
    numberOfPayments,
    paymentAmount: paymentAmounts[0],           // most payments
    finalPaymentAmount: paymentAmounts[numberOfPayments - 1],
    paymentDates,
    frequency,
    startDate: new Date(validStart),
    endDate: new Date(validEnd)
  };
}

// Improved biweekly checkout - now uses same even distribution
export function calculateCheckoutBiweeklySchedule(
  totalAmountCents: number,
  classStartDate: Date,
  classEndDate: Date,
  anchorDate?: Date
): any {
  const now = anchorDate || new Date();
  const scheduleStart = classStartDate > now ? classStartDate : now;
  const schedule = calculatePaymentSchedule(totalAmountCents, scheduleStart, classEndDate, 'biweekly');

  if (schedule.numberOfPayments === 1) {
    return { firstPaymentAmount: totalAmountCents, numberOfPayments: 1, paymentAmount: totalAmountCents, finalPaymentAmount: totalAmountCents, paymentDates: [now], totalAmount: totalAmountCents };
  }

  const allDates = classStartDate > now ? [now, ...schedule.paymentDates] : schedule.paymentDates;
  const totalPayments = allDates.length;
  const base = Math.floor(totalAmountCents / totalPayments);
  const remainder = totalAmountCents - (base * totalPayments);
  const paymentAmounts = distributeRemainder(base, remainder, totalPayments);

  return {
    firstPaymentAmount: paymentAmounts[0],
    numberOfPayments: totalPayments,
    paymentAmount: paymentAmounts[0],
    finalPaymentAmount: paymentAmounts[totalPayments - 1],
    paymentDates: allDates,
    totalAmount: totalAmountCents
  };
}
