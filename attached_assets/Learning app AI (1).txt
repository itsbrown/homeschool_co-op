const express = require('express');
const { spawn } = require('child_process');
const path = require('path');
const PDFDocument = require('pdfkit');
const AWS = require('aws-sdk');
const axios = require('axios');
const s3 = new AWS.S3();
const router = express.Router();
const Activity = require('../models/activity');
const activityQueue = require('../queues/activityQueue');


router.post('/generate', auth(['admin', 'educator']), async (req, res) => {
  const { age, subject, activityType } = req.body;
  if (!age || !subject || !activityType) {
    return res.status(400).json({ message: 'Age, subject, and activity type are required' });
  }
  const subjectMap = { 'math': 'Mathematics', 'science': 'Science', 'history': 'History' };
  const mappedSubject = subjectMap[subject.toLowerCase()] || subject;


  const job = await activityQueue.add({ age, subject: mappedSubject, activityType });
  job.on('failed', (err) => {
    res.status(500).json({ message: 'Activity generation failed', error: err });
  });
  job.on('completed', async (result) => {
    const activity = result;
    const doc = new PDFDocument({ size: 'A4', margin: 20 });
    let buffers = [];
    doc.on('data', buffers.push.bind(buffers));
    doc.on('end', async () => {
      const pdfData = Buffer.concat(buffers);
      const key = `activities/${activityType}-${Date.now()}.pdf`;
      await s3.upload({
        Bucket: process.env.AWS_S3_BUCKET,
        Key: key,
        Body: pdfData,
        ContentType: 'application/pdf',
      }).promise();
      const downloadLink = s3.getSignedUrl('getObject', {
        Bucket: process.env.AWS_S3_BUCKET,
        Key: key,
        Expires: 3600,
      });
      const activityRecord = new Activity({
        type: activityType,
        url: downloadLink,
        subject: mappedSubject,
        ageRange: age,
      });
      await activityRecord.save();
      await axios.post('http://localhost:3000/api/email/send', {
        email: 'user@example.com',
        subject: 'Activity Generated',
        message: `Your ${activityType} is ready: ${downloadLink}`,
      });
    });


    // Child-friendly PDF design
    doc.font('Helvetica-Bold').fontSize(20).text("America's Birthday: Founding Heroes", 50, 30, { align: 'center' });
    doc.font('Helvetica').fontSize(12).text('Colored by: ___________________', 50, 60, { align: 'center' });


    // Draw a fun border
    doc.lineWidth(2).rect(20, 20, 555, 802).stroke();


    // Instructions
    doc.fontSize(14).text('Instructions:', 50, 90);
    doc.fontSize(12).text('• Color the different elements of America’s founding story.', 50, 110);
    doc.text('• Read the learning facts with an adult to learn more!', 50, 130);


    // Coloring Guide
    let yPosition = 160;
    doc.fontSize(14).text('Coloring Guide:', 50, yPosition);
    yPosition += 20;
    activity.coloring_guide.forEach(item => {
      doc.fontSize(12).text(`• ${item.element}: ${item.instruction}`, 50, yPosition);
      yPosition += 20;
    });


    // Generate and embed the coloring image using Stable Diffusion (via SageMaker)
    let imageData;
    try {
      const imageResponse = await axios.post('https://your-sagemaker-endpoint/invocations', {
        prompt: 'A simplified cartoon-style scene for young children showing George Washington in his general’s uniform, the Liberty Bell with a crack, an American flag with 13 stars in a circle, and Independence Hall in Philadelphia, black-and-white line art',
        negative_prompt: 'color, shading, realistic, complex details',
        steps: 50,
        cfg_scale: 7.5,
      }, { responseType: 'arraybuffer' });
      imageData = imageResponse.data;


      // Upload the generated image to S3 for future use
      const imageKey = `coloring-images/history-${Date.now()}.png`;
      await s3.upload({
        Bucket: process.env.AWS_S3_BUCKET,
        Key: imageKey,
        Body: imageData,
        ContentType: 'image/png',
      }).promise();
    } catch (err) {
      console.error('Image generation failed:', err);
      doc.fontSize(12).text('Coloring image unavailable. Please try again later.', 50, yPosition + 20);
      doc.end();
      return;
    }


    // Dynamically adjust image position and size to fit within the page
    const pageHeight = 842; // A4 height in points
    const imageHeight = 300;
    const factsHeight = activity.facts.length * 20 + 40; // Estimate facts section height
    const remainingSpace = pageHeight - yPosition - factsHeight - 60; // Leave space for facts and margins
    const adjustedImageHeight = Math.min(imageHeight, remainingSpace);
    doc.image(imageData, 50, yPosition + 20, { width: 500, height: adjustedImageHeight });


    // Did You Know? Section
    yPosition += adjustedImageHeight + 40;
    doc.fontSize(14).text('Did You Know?', 50, yPosition);
    yPosition += 20;
    activity.facts.forEach(fact => {
      doc.fontSize(12).text(`• ${fact}`, 50, yPosition);
      yPosition += 20;
    });


    doc.end();
  });
  res.json({ message: 'Activity generation queued', jobId: job.id });
});
