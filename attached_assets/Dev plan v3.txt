import sys
import json
from transformers import pipeline


def generate_curriculum(profile):
    nlp = pipeline('text-generation', model='distilgpt2')
    files = [f for base in profile['bases'] for f in base['files']]
    prompt = f"Generate a curriculum for a {profile['age']} year old using files: {', '.join(f['url'] for f in files)}"
    curriculum = nlp(prompt, max_length=100)[0]['generated_text']
    return {'curriculum': curriculum, 'lessons': [{'id': 1, 'title': 'Intro', 'content': files[0]['url'] if files else 'Sample'}]}

import React, { useState } from 'react';
import { useQuery, gql } from '@apollo/client';
import Card from './Card';


const GET_BASES = gql`query { knowledgeBases { id title subject } }`;


const KnowledgeBaseSelector = ({ onSelect }) => {
  const { loading, data } = useQuery(GET_BASES);
  const [selectedIds, setSelectedIds] = useState([]);


  const handleSelect = (id) => {
    const newIds = selectedIds.includes(id)
      ? selectedIds.filter(sid => sid !== id)
      : [...selectedIds, id];
    setSelectedIds(newIds);
    onSelect(newIds);
  };


  return (
    <Card className="p-4">
      <h2 className="text-xl font-bold text-primary mb-2">Select Knowledge Bases</h2>
      {loading ? <p>Loading...</p> : data.knowledgeBases.map(base => (
        <div key={base.id} className="flex items-center mb-2">
          <input
            type="checkbox"
            checked={selectedIds.includes(base.id)}
            onChange={() => handleSelect(base.id)}
            className="mr-2"
          />
          <span>{base.title} ({base.subject})</span>
        </div>
      ))}
    </Card>
  );
};
