const express = require('express');
const { spawn } = require('child_process');
const path = require('path');
const PDFDocument = require('pdfkit');
const AWS = require('aws-sdk');
const axios = require('axios');
const s3 = new AWS.S3();
const router = express.Router();
const Activity = require('../models/activity');
const activityQueue = require('../queues/activityQueue');


router.post('/generate', auth(['admin', 'educator']), async (req, res) => {
  const { age, subject, activityType } = req.body;
  if (!age || !subject || !activityType) {
    return res.status(400).json({ message: 'Age, subject, and activity type are required' });
  }
  const subjectMap = { 'math': 'Mathematics', 'science': 'Science', 'history': 'History' };
  const mappedSubject = subjectMap[subject.toLowerCase()] || subject;


  const job = await activityQueue.add({ age, subject: mappedSubject, activityType });
  job.on('failed', (err) => {
    res.status(500).json({ message: 'Activity generation failed', error: err });
  });
  job.on('completed', async (result) => {
    const activity = result;
    const doc = new PDFDocument({ size: 'A4', margin: 20 });
    let buffers = [];
    doc.on('data', buffers.push.bind(buffers));
    doc.on('end', async () => {
      const pdfData = Buffer.concat(buffers);
      const key = `activities/${activityType}-${Date.now()}.pdf`;
      await s3.upload({
        Bucket: process.env.AWS_S3_BUCKET,
        Key: key,
        Body: pdfData,
        ContentType: 'application/pdf',
      }).promise();
      const downloadLink = s3.getSignedUrl('getObject', {
        Bucket: process.env.AWS_S3_BUCKET,
        Key: key,
        Expires: 3600,
      });
      const activityRecord = new Activity({
        type: activityType,
        url: downloadLink,
        subject: mappedSubject,
        ageRange: age,
      });
      await activityRecord.save();
      await axios.post('http://localhost:3000/api/email/send', {
        email: 'user@example.com',
        subject: 'Activity Generated',
        message: `Your ${activityType} is ready: ${downloadLink}`,
      });
    });


    // Child-friendly PDF design
    doc.font('Helvetica-Bold').fontSize(20).text('American Founding Symbols Coloring Adventure', 50, 30, { align: 'center' });
    doc.font('Helvetica').fontSize(12).text('Colored by: ___________________', 50, 60, { align: 'center' });


    // Draw a fun border
    doc.lineWidth(2).rect(20, 20, 555, 802).stroke();


    // Instructions
    doc.fontSize(14).text('Instructions:', 50, 90);
    doc.fontSize(12).text('• Color each part of the picture as suggested or use your own colors!', 50, 110);
    doc.text('• Think about what each symbol means for our country.', 50, 130);


    // Coloring Guide
    doc.fontSize(14).text('Coloring Guide:', 50, 160);
    doc.fontSize(12).text('• Liberty Bell: Color it brown.', 50, 180);
    doc.text('• American Flag: Stripes red and white, stars blue with white stars.', 50, 200);
    doc.text('• Bald Eagle: Body brown, head and tail white.', 50, 220);
    doc.text('• Constitution: Light yellow or tan to look old.', 50, 240);
    doc.text('• Independence Hall: Red bricks, white windows.', 50, 260);


    // Fetch and embed the coloring image from S3
    const imageUrl = 'https://your-bucket.s3.amazonaws.com/coloring-images/american_founding_symbols.png';
    const imageResponse = await axios.get(imageUrl, { responseType: 'arraybuffer' });
    doc.image(imageResponse.data, 50, 290, { width: 500, height: 300 });


    // Did You Know? Section
    doc.fontSize(14).text('Did You Know?', 50, 610);
    doc.fontSize(12).text('• The Liberty Bell is a symbol of American independence.', 50, 630);
    doc.text('• The American flag has 13 stripes for the original colonies.', 50, 650);
    doc.text('• The bald eagle represents strength and freedom.', 50, 670);
    doc.text('• The Constitution begins with "We the People".', 50, 690);
    doc.text('• Independence Hall is where the Declaration was signed.', 50, 710);


    doc.end();
  });
  res.json({ message: 'Activity generation queued', jobId: job.id });
});
