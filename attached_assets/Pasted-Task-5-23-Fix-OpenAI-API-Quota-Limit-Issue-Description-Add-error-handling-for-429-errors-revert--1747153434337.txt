import time
import requests

def call_openai_api(data, retries=3):
    for i in range(retries):
        try:
            response = requests.post('https://api.openai.com/v1/completions', 
                             headers={'Authorization': f'Bearer {os.getenv("OPENAI_API_KEY")}'},
                             json={'prompt': f"Generate a {data['activityType']} for {data['subject']}", 'max_tokens': 100})
            if response.status_code == 429:
                delay = 2 ** i
                time.sleep(delay)
                continue
            response.raise_for_status()
            return response.json()
        except requests.exceptions.RequestException as e:
            if i == retries - 1:
                raise Exception(f"Failed to call OpenAI API after {retries} retries: {str(e)}")
    return None
