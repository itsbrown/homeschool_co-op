Add deterministic split utility
File: server/lib/splitIntegerEvenly.ts
Purpose: split totalCents into N integer shares that sum exactly to totalCents
Template (use exactly or equivalent)

ts
// server/lib/splitIntegerEvenly.ts
export function splitIntegerEvenly(total: number, parts: number): number[] {
  if (!Number.isInteger(total) || total < 0) throw new Error('total must be non-negative integer');
  if (!Number.isInteger(parts) || parts <= 0) throw new Error('parts must be positive integer');

  const base = Math.floor(total / parts);
  let remainder = total % parts;
  const result: number[] = [];
  for (let i = 0; i < parts; i++) {
    result.push(remainder > 0 ? base + 1 : base);
    if (remainder > 0) remainder -= 1;
  }
  const sum = result.reduce((s, v) => s + v, 0);
  if (sum !== total) throw new Error(`split mismatch: ${sum} !== ${total}`);
  return result;
}

export type CartItemInput = { classId: number; childId?: number; quantity?: number; /* client hints ignored */ };

export function canonicalizeItems(items: CartItemInput[], lookupPrice: (classId:number)=>number) {
  return items.map(it => {
