Create /server/src/routes/auth.js:
 javascript
Copy
const express = require('express');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
const User = require('../models/user');
const router = express.Router();

router.post('/register', async (req, res) => {
  const { email, password, role } = req.body;
  const hashedPassword = await bcrypt.hash(password, 10);
  const user = new User({ email, password: hashedPassword, role });
  await user.save();
  res.status(201).json({ message: 'User registered' });
});

router.post('/login', async (req, res) => {
  const { email, password } = req.body;
  const user = await User.findOne({ email });
  if (!user || !await bcrypt.compare(password, user.password)) {
    return res.status(401).json({ message: 'Invalid credentials' });
  }
  const token = jwt.sign({ userId: user._id, role: user.role }, process.env.JWT_SECRET, { expiresIn: '1h' });
  res.json({ token });
});

module.exports = router;

const path = require('path');

const generateCurriculum = (profile) => new Promise((resolve, reject) => {
  const python = spawn('python', [path.join(__dirname, 'curriculum.py'), JSON.stringify(profile)]);
  let output = '';
  python.stdout.on('data', (data) => output += data);
  python.stderr.on('data', (data) => reject(data.toString()));
  python.on('close', () => resolve(JSON.parse(output)));
});

Copy
import sys
import json
from transformers import pipeline

def generate_curriculum(profile):
    nlp = pipeline('text-generation', model='distilgpt2')
    prompt = f"Generate a curriculum for a {profile['age']} year old student interested in {profile['subjects']}"
    curriculum = nlp(prompt, max_length=100)[0]['generated_text']
    return {'curriculum': curriculum, 'lessons': [{'id': 1, 'title': 'Intro Lesson', 'content': 'Sample content'}]}

if __name__ == '__main__':
    profile = json.loads(sys.argv[1])
    result = generate_curriculum(profile)
    print(json.dumps(result))

import React, { useState } from 'react';
import axios from 'axios';
import Card from './Card';

const TutorChat = ({ lessonId }) => {
  const [question, setQuestion] = useState('');
  const [response, setResponse] = useState('');

  const handleSubmit = async () => {
    const res = await axios.post('http://localhost:3000/api/curriculum/tutor', {
      question,
      context: `Lesson ${lessonId}`,
    });
    setResponse(res.data.response);
  };

  return (
    <Card className="p-4">
      <input
        type="text"
        value={question}
        onChange={(e) => setQuestion(e.target.value)}
        className="border p-2 w-full mb-2"
        placeholder="Ask the tutor..."
      />
      <button onClick={handleSubmit} className="bg-primary text-white px-4 py-2 rounded-md">Ask</button>
      {response && <p className="mt-2">{response}</p>}
    </Card>
  );
};
export default TutorChat;

const KnowledgeBase = require('./models/knowledgeBase');
const CommunityPost = require('./models/communityPost');

const resolvers = {
  Query: {
    knowledgeBases: async () => await KnowledgeBase.find(),
    communityPosts: async () => await CommunityPost.find(),
  },
  Mutation: {
    createKnowledgeBase: async (_, { title, creator, subject, price }) => {
      const base = new KnowledgeBase({ title, creator, subject, price });
      await base.save();
      return base;
    },
    createCommunityPost: async (_, { content, author }) => {
      const post = new CommunityPost({ content, author });
      await post.save();
      return post;
    },
  },
};
module.exports = resolvers;

import React, { useState, useEffect } from 'react';
import io from 'socket.io-client';
import Card from './Card';

const socket = io('http://localhost:3000');

const Chat = () => {
  const [message, setMessage] = useState('');
  const [posts, setPosts] = useState([]);

  useEffect(() => {
    socket.on('newPost', (post) => setPosts([...posts, post]));
    return () => socket.off('newPost');
  }, [posts]);

  const sendMessage = () => {
    socket.emit('post', { content: message, author: 'User' });
    setMessage('');
  };

  return (
    <Card className="p-4">
      <div className="h-64 overflow-y-scroll mb-2">
        {posts.map((post, i) => <p key={i}>{post.author}: {post.content}</p>)}
      </div>
      <input
        type="text"
        value={message}
        onChange={(e) => setMessage(e.target.value)}
        className="border p-2 w-full mb-2"
      />
      <button onClick={sendMessage} className="bg-primary text-white px-4 py-2 rounded-md">Send</button>
    </Card>
  );
};
